104 __STATIC_INLINE app_addr_t getUniqueAddress()
105 {
106  app_addr_t address = getUniqueId();
107 
108  // Check the address is in correct range
109  if (address == 0x0)
110  {
111  // Zero is not a valid address
112  address = 1;
113  }
114  else if ((address & 0xFF000000) == 0x80000000)
115  {
116  // Address would be in the multicast space
117  address &= 0x7ffffffd;
118  }
119 
120  // Workaround:
121  // there is a bug in Wirepas reference gateway fixed in version 1.3.
122  // It doesn't forward messages from nodes with an address with the
123  // MSB set to one.
124  // In order to avoid this situation, explicitly reset the highest bit when
125  // generating a unique address.
126  // If the target network only has gateways with version >= 1.3, or with different
127  // gateway implementation it is safe to remove this line
128  address &= 0x7fffffff;
129 
130  return address;
131 }
