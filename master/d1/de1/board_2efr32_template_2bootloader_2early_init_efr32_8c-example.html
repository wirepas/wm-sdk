#include "early_init.h"
#include "mcu.h"
//#define MCU_NO_DCDC
void early_init(void)
{
volatile uint32_t x;
x = EMU->EM4CTRL;
#ifdef MCU_NO_DCDC
x |= (1 << _EMU_EM4CTRL_EM4IORETMODE_SHIFT); // 1
#else
x &= ~(_EMU_EM4CTRL_EM4IORETMODE_MASK); // 0
#endif
EMU->EM4CTRL = x;
}
